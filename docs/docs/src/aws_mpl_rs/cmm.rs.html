106///Inputs for creating an Required Encryption Context Cryptographic Materials Manager.
107#[derive(Clone, Debug, Default)]
108#[non_exhaustive]
109pub struct CreateRequiredEncryptionContextCMMInput {
110    ///The Cryptographic Materials Manager that the created Required Encryption Context Cryptographic Materials Manager will delegate to. Either a Keyring or underlying Cryptographic Materials Manager must be specified.
111    pub underlying_cmm: Option<CryptographicMaterialsManagerRef>,
112    ///The Keyring that the created Cryptographic Materials Manager will use to wrap data keys. The created Required Encryption Context CMM will delegate to a Default Cryptographic Materials Manager created with this Keyring. Either a Keyring or an underlying Cryptographic Materials Manager must be specified as input.
113    pub keyring: Option<KeyringRef>,
114
115    ///A list of Encryption Context keys which are required to be supplied during encryption and decryption, and correspond to Encryption Context key-value pairs which are not stored on the resulting message.
116    pub required_encryption_context_keys: Vec<String>,
117}
118
119impl CreateRequiredEncryptionContextCMMInput {
120    pub(crate) fn validate(&self) -> Result<(), Error> {
121        if self.underlying_cmm.is_none() && self.keyring.is_none() {
122            return Err(mpl_err(
123                "Either a Keyring or underlying Cryptographic Materials Manager must be specified."
124                    .to_string(),
125            ));
126        }
127        if self.underlying_cmm.is_some() && self.keyring.is_some() {
128            return Err(mpl_err(
129                "Only one of Keyring or underlying Cryptographic Materials Manager must be specified."
130                    .to_string(),
131            ));
132        }
133        Ok(())
134    }
135
136    pub fn with_keyring<T: Into<String> + Clone>(keyring: KeyringRef, keys: &[T]) -> Self {
137        Self {
138            underlying_cmm: None,
139            keyring: Some(keyring),
140            required_encryption_context_keys: keys.iter().map(|k| k.clone().into()).collect(),
141        }
142    }
143    pub fn with_cmm<T: Into<String> + Clone>(
144        cmm: CryptographicMaterialsManagerRef,
145        keys: &[T],
146    ) -> Self {
147        Self {
148            underlying_cmm: Some(cmm),
149            keyring: None,
150            required_encryption_context_keys: keys.iter().map(|k| k.clone().into()).collect(),
151        }
152    }
153    pub fn go(&self) -> Result<CryptographicMaterialsManagerRef, Error> {
154        create_required_encryption_context_cmm(self)
155    }
