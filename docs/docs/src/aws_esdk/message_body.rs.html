12#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
13pub(crate) enum BodyAADContent {
14    RegularFrame,
15    FinalFrame,
16    SingleBlock,
17}
18
19const BODY_AAD_CONTENT_REGULAR_FRAME: &str = "AWSKMSEncryptionClient Frame";
20const BODY_AAD_CONTENT_FINAL_FRAME: &str = "AWSKMSEncryptionClient Final Frame";
21const BODY_AAD_CONTENT_SINGLE_BLOCK: &str = "AWSKMSEncryptionClient Single Block";
22
23const fn body_aad_content_type_string(bc: BodyAADContent) -> &'static str {
24    match bc {
25        BodyAADContent::RegularFrame => BODY_AAD_CONTENT_REGULAR_FRAME,
26        BodyAADContent::FinalFrame => BODY_AAD_CONTENT_FINAL_FRAME,
27        BodyAADContent::SingleBlock => BODY_AAD_CONTENT_SINGLE_BLOCK,
28    }
29}

46//# The IV length MUST be equal to the IV length of the [algorithm suite](../framework/algorithm-suites.md) that generated the message.
47pub(crate) fn iv_seq(sequence_number: u32, result: &mut [u8]) {
48    let pivot = result.len() - 4;
49    result[pivot..].copy_from_slice(&sequence_number.to_be_bytes());
50}

52pub(crate) fn get_encrypt(info: &AlgorithmSuite) -> AesGcm {
53    match &info.encrypt {
54        aws_mpl_rs::suites::Encrypt::AesGcm(aes_gcm) => *aes_gcm,
55        _ => panic!("not an aes gcm"),
56    }
57}

60 * Serializes the Message Body ADD

63pub(crate) fn body_aad(
64    message_id: &[u8],
65    bc: BodyAADContent,
66    sequence_number: u32,
67    length: u64,
68    result: &mut Vec<u8>,
69) {
70    result.clear();
71    result.extend_from_slice(message_id);
72    result.extend_from_slice(body_aad_content_type_string(bc).as_bytes());
73    result.extend_from_slice(&sequence_number.to_be_bytes());
74    result.extend_from_slice(&length.to_be_bytes());
75}

77pub(crate) fn read_and_decrypt_framed_message_body(
78    r: &mut dyn SafeRead,
79    w: &mut dyn SafeWrite,
80    header: &HeaderInfo,
81    key: &[u8],
82    raw: &mut dyn SafeWrite,
83    fail_if_multi_frame: bool,
84) -> Result<Vec<u8>, Error> {
85    let mut expected_frame: u32 = START_SEQUENCE_NUMBER;
86    let mut iv = vec![0u8; get_iv_length(&header.suite) as usize];
87    let mut auth_tag = vec![0u8; get_tag_length(&header.suite) as usize];
88    let alg = get_encrypt(&header.suite);
89    let frame_length_u64 = u64::from(header.body.frame_length());
90    let frame_length_usize = header.body.frame_length() as usize;
91    let mut enc_content = vec![0u8; frame_length_usize];
92    let mut result = vec![0; frame_length_usize];
93    let mut aad = Vec::new();
94
95    loop {
96        let seq_num = read_u32(r, raw)?;
97        if seq_num == ENDFRAME_SEQUENCE_NUMBER {
98            let seq_num: u32 = read_u32(r, raw)?;
99            if seq_num != expected_frame {
100                return Err("Final sequence number out of order.".into());
101            }
102            read_bytes(r, &mut iv, raw)?;
103            read_seq_u32_bounded(
104                r,
105                header.body.frame_length(),
106                "Content length MUST NOT exceed the frame length.",
107                &mut enc_content,
108                raw,
109            )?;
110            read_bytes(r, &mut auth_tag, raw)?;
111            body_aad(
112                header.body.message_id(),
113                BodyAADContent::FinalFrame,
114                seq_num,
115                enc_content.len() as u64,
116                &mut aad,
117            );
118            if enc_content.is_empty() {
119                // final frame is empty, so return last full frame
120                let mut empty_result = Vec::new();
121                aes_decrypt(
122                    alg,
123                    key,
124                    &enc_content,
125                    &auth_tag,
126                    &iv,
127                    &aad,
128                    &mut empty_result[..],
129                )?;
130            } else {
131                // write previous frame's data, now that we know we have another frame.
132                if expected_frame != START_SEQUENCE_NUMBER {
133                    if fail_if_multi_frame {
134                        return Err("Streaming Interface can return data before signature has been validated. Set `i_accept_the_danger` in the DecryptStreamInput struct if this is ok.".into());
135                    }
136                    serialize_functions::write_bytes(w, &result)?;
137                }
138                aes_decrypt(
139                    alg,
140                    key,
141                    &enc_content,
142                    &auth_tag,
143                    &iv,
144                    &aad,
145                    &mut result[0..enc_content.len()],
146                )?;
147                result.resize(enc_content.len(), 0);
148            }
149            return Ok(result);
150        }
151        if seq_num != expected_frame {
152            return Err("Sequence number out of order.".into());
153        }
154        // write previous frame's data, now that we know we have another frame.
155        if expected_frame != START_SEQUENCE_NUMBER {
156            if fail_if_multi_frame {
157                return Err("Streaming Interface can return data before signature has been validated. Set `i_accept_the_danger` in the DecryptStreamInput struct if this is ok.".into());
158            }
159            serialize_functions::write_bytes(w, &result)?;
160        }
161        expected_frame += 1;
162        read_bytes(r, &mut iv, raw)?;
163        read_bytes(r, &mut enc_content, raw)?;
164        read_bytes(r, &mut auth_tag, raw)?;
165        body_aad(
166            header.body.message_id(),
167            BodyAADContent::RegularFrame,
168            seq_num,
169            frame_length_u64,
170            &mut aad,
171        );
172        aes_decrypt(alg, key, &enc_content, &auth_tag, &iv, &aad, &mut result)?;
173    }
174}
