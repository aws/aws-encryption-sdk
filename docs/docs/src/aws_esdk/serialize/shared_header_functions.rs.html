10pub(crate) fn read_esdk_suite_id(
11    r: &mut dyn SafeRead,
12    raw: &mut dyn SafeWrite,
13) -> Result<&'static AlgorithmSuite, Error> {
14    let mut esdk_suite_id_bytes = [0; 2];
15    read_bytes(r, &mut esdk_suite_id_bytes, raw)?;
16
17    let suite = aws_mpl_rs::suites::get_algorithm_suite_info(esdk_suite_id_bytes)?;
18    Ok(suite)
19}

49/*
50 * Writes the message id as bytes, which, since the message id is already stored
51 * as bytes, simply returns the message id.
52 *
53 * Though we have different V1 and V2 methods for the read path, since
54 * they read different numbers of bytes, a single method on the write path
55 * is fine since writing is identical for both.
56 */
57pub(crate) fn write_message_id(w: &mut dyn SafeWrite, message_id: &MessageId) -> Result<(), Error> {
58    write_bytes(w, message_id)
59}
