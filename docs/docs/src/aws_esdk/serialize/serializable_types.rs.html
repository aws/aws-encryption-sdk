14pub(crate) const fn get_iv_length(a: &AlgorithmSuite) -> u8 {
15    match a.encrypt {
16        aws_mpl_rs::suites::Encrypt::AesGcm(_e) => 12,
17        _ => 0,
18    }
19}

21pub(crate) const fn get_tag_length(a: &AlgorithmSuite) -> u8 {

35/*
36 * Length properties of the Encryption Context.
37 * The Encryption Context has a complex relationship with length.
38 * Each key or value MUST be less than Uint16,
39 * However the entire thing MUST also serialize to less than Uint16.
40 * In practice, this means than the longest value,
41 * given a key of 1 bytes is Uint16-2-2-1.
42 * e.g.
43 * 2 for the key length
44 * 1 for the key data
45 * 2 for the value length
46 * Uint16-2-2-1 for the value data
47 */
48
49pub(crate) fn length(encryption_context: &ESDKEncryptionContext) -> u64 {
50    let mut length: usize = 0;
51    for (key, value) in encryption_context {
52        length += 2 + key.len() + 2 + value.len();
53    }
54    length as u64
55}

57pub(crate) fn to_canonical_pairs(
58    encryption_context: ESDKEncryptionContext,
59) -> ESDKCanonicalEncryptionContext {
60    let mut pairs: Vec<(String, String)> = encryption_context.into_iter().collect();
61    pairs.sort_by(|a, b| a.0.cmp(&b.0));
62    pairs
63}
64
65pub(crate) fn from_canonical_pairs(pairs: ESDKCanonicalEncryptionContext) -> ESDKEncryptionContext {
66    let mut map: ESDKEncryptionContext = ESDKEncryptionContext::new();
67    for (key, value) in pairs {
68        map.insert(key, value);
69    }
70    map
71}

73pub(crate) fn is_esdk_encryption_context(ec: &EncryptionContext) -> bool {
74    if ec.len() >= u16::MAX as usize {
75        return false;
76    }
77    if length(ec) >= ESDK_CANONICAL_ENCRYPTION_CONTEXT_MAX_LENGTH {
78        return false;
79    }
80    for (key, value) in ec {
81        if key.len() >= u16::MAX as usize {
82            return false;
83        }
84        if value.len() >= u16::MAX as usize {
85            return false;
86        }
87    }
88    true
89}

91pub(crate) fn is_esdk_encrypted_data_key(edk: &EncryptedDataKey) -> bool {
92    u16::try_from(edk.key_provider_id.len()).is_ok()
93        && u16::try_from(edk.key_provider_info.len()).is_ok()
94}
95
96pub(crate) fn is_esdk_encrypted_data_keys(edks: &[EncryptedDataKey]) -> bool {
97    if edks.len() >= u16::MAX as usize {
98        return false;
99    }
100    for edk in edks {
101        if !is_esdk_encrypted_data_key(edk) {
102            return false;
103        }
104    }
105    true
106}
