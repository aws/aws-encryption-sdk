14//= ../specification/data-format/message-header.md#reserved
15//= type=implication
16//# A reserved sequence of 4 bytes
17//# that MUST have the value (hex) of `00 00 00 00`.
18const RESERVED_BYTES: [u8; 4] = [0x00, 0x00, 0x00, 0x00];
19
20pub(crate) fn write_v1_header_body(
21    w: &mut dyn SafeWrite,
22    body: &V1HeaderBody,
23) -> Result<(), Error> {
24    //= ../specification/data-format/message-header.md#header-body-version-1-0
25    //= type=implication
26    //# The value of the `Version` field MUST be `01` in the Version 1.0 header body.
27
28    //= ../specification/data-format/message-header.md#version-1
29    //= type=implication
30    //# The version (hex) of this field MUST be a value that exists in the following table:
31    write_msg_format_version(w, MessageFormatVersion::V1)?;
32    //= ../specification/data-format/message-header.md#type
33    //= type=implication
34    //# The type (hex) of this field MUST be a value that exists in the following table:
35    write_msg_type(w, body.message_type)?;
36    write_esdk_suite_id(w, &body.algorithm_suite)?;
37    write_message_id(w, &body.message_id)?;
38    write_aad_section(w, &body.encryption_context)?;
39    write_edks(w, &body.encrypted_data_keys)?;
40    write_content_type(w, body.content_type)?;
41    write_bytes(w, &RESERVED_BYTES)?;
42    //= ../specification/data-format/message-header.md#iv-length
43    //= type=implication
44    //# This value MUST be equal to the [IV length](../framework/algorithm-suites.md#iv-length) value of the
45    //# [algorithm suite](../framework/algorithm-suites.md) specified by the [Algorithm Suite ID](#algorithm-suite-id) field.
46    write_u8(w, get_iv_length(&body.algorithm_suite))?;
47    write_u32(w, body.frame_length)
48}

50pub(crate) fn read_v1_reserved_bytes(
51    r: &mut dyn SafeRead,
52    raw: &mut dyn SafeWrite,
53) -> Result<(), Error> {
54    let mut result = [0; RESERVED_BYTES.len()];
55    read_bytes(r, &mut result, raw)?;
56    if result == RESERVED_BYTES {
57        Ok(())
58    } else {
59        ser_err("Incorrect reserved bytes.")
60    }
61}

63pub(crate) fn read_v1_header_iv_length(
64    r: &mut dyn SafeRead,
65    suite: &AlgorithmSuite,
66    raw: &mut dyn SafeWrite,
67) -> Result<u8, Error> {
68    let raw = read_u8(r, raw)?;
69    if raw == get_iv_length(suite) {
70        Ok(raw)
71    } else {
72        ser_err("HeaderIv Length does not match Algorithm Suite.")
73    }
74}

76pub(crate) fn read_v1_header_body(
77    r: &mut dyn SafeRead,
78    max_edks: Option<std::num::NonZeroUsize>,
79    raw: &mut dyn SafeWrite,
80) -> Result<V1HeaderBody, Error> {
81    let message_type = read_msg_type(r, raw)?;
82    let algorithm_suite = read_esdk_suite_id(r, raw)?;
83    let message_id = read_message_id_v1(r, raw)?;
84    let encryption_context: Vec<(String, String)> = read_canonical_ec(r, raw)?;
85    let encrypted_data_keys = read_edks(r, max_edks, raw)?;
86    let content_type = read_content_type(r, raw)?;
87    read_v1_reserved_bytes(r, raw)?;
88    let header_iv_length = read_v1_header_iv_length(r, algorithm_suite, raw)?;
89    let frame_length = read_u32(r, raw)?;
90
91    Ok(V1HeaderBody {
92        message_type,
93        algorithm_suite: algorithm_suite.clone(),
94        message_id,
95        encryption_context,
96        encrypted_data_keys,
97        content_type,
98        header_iv_length: u64::from(header_iv_length),
99        frame_length,
100    })
