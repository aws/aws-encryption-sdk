102/// Secrets, e.g. Plain text keys
103#[allow(clippy::exhaustive_structs)]
104pub struct Secret(pub Vec<u8>);
105
106impl Debug for Secret {
107    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
108        write!(f, "Secret of size {}", self.0.len())
109    }
110}
111
112impl Secret {
113    #[must_use]
114    pub const fn new(bytes: Vec<u8>) -> Self {
115        Self(bytes)
116    }
117
118    #[must_use]
119    pub fn as_bytes(&self) -> &[u8] {
120        &self.0
121    }
122}
123
124impl Drop for Secret {
125    fn drop(&mut self) {
126        self.0.zeroize();
127    }
128}

213//= aws-encryption-sdk-specification/framework/structures.md#structure
214//= type=implication
215//# An encrypted data key is comprised of the following fields:
216//#
217//# - [Key Provider ID](#key-provider-id)
218//# - [Key Provider Information](#key-provider-information)
219//# - [Ciphertext](#ciphertext)
220//#
221//# Note: "Encrypted" is a misnomer here, as the process by which a key provider may obtain the plaintext data key
222//# from the ciphertext and vice versa does not have to be an encryption and decryption cipher.
223//# This specification uses the terms "encrypt" and "decrypt" for simplicity,
224//# but the actual process by which a key provider obtains the plaintext data key from the ciphertext
225//# and vice versa MAY be any reversible operation, though we expect that most will use encryption.
226#[derive(Debug, PartialEq, Eq, Clone, Default, Hash)]
227#[non_exhaustive]
228pub struct EncryptedDataKey {
229    // The spec defines keyProviderId in 2 places,
230    // and, while they are not identical,
231    // they do not disagree.
232    // data-format/message-header.md#encrypted-data-key-entries ::
233    // UTF-8 encoded bytes
234    // framework/keyring-interface.md#key-provider-id ::
235    // The key provider ID MUST be a binary value and SHOULD be equal to a UTF-8 encoding of the key namespace.
236    pub key_provider_id: String,
237
238    // The key provider info MUST be a binary value and SHOULD be equal to a UTF-8 encoding of the key name.
239    pub key_provider_info: Vec<u8>,
240    pub ciphertext: Vec<u8>,
241}
242
243impl EncryptedDataKey {
244    pub fn new(
245        key_provider_id: impl Into<String>,
246        key_provider_info: impl Into<Vec<u8>>,
247        ciphertext: impl Into<Vec<u8>>,
248    ) -> Self {
249        Self {
250            key_provider_id: key_provider_id.into(),
251            key_provider_info: key_provider_info.into(),
252            ciphertext: ciphertext.into(),
253        }
254    }
