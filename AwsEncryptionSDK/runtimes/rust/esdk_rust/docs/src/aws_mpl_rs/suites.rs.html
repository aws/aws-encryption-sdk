117#[derive(Clone, Copy, Debug, Default, Eq, PartialEq)]

119pub enum DerivationAlgorithm {
120    Hkdf(Hkdf),
121    // We are using both `IDENTITY` and `None` here
122    // to model the fact that deriving
123    // the data encryption key and the commitment key
124    // MUST be the same.
125    // The specification treats NO_KDF as an identity operation.
126    // So this naming convention mirrors the specification.
127    Identity,
128    #[default]
129    None,
130}

203pub fn get_algorithm_suite_info(binary_id: [u8; 2]) -> Result<&'static AlgorithmSuite, Error> {
204    match binary_id {
205        [0x00, 0x14] => Ok(&ESDK_ALG_AES_128_GCM_IV12_TAG16_NO_KDF),
206        [0x00, 0x46] => Ok(&ESDK_ALG_AES_192_GCM_IV12_TAG16_NO_KDF),
207        [0x00, 0x78] => Ok(&ESDK_ALG_AES_256_GCM_IV12_TAG16_NO_KDF),
208        [0x01, 0x14] => Ok(&ESDK_ALG_AES_128_GCM_IV12_TAG16_HKDF_SHA256),
209        [0x01, 0x46] => Ok(&ESDK_ALG_AES_192_GCM_IV12_TAG16_HKDF_SHA256),
210        [0x01, 0x78] => Ok(&ESDK_ALG_AES_256_GCM_IV12_TAG16_HKDF_SHA256),
211        [0x02, 0x14] => Ok(&ESDK_ALG_AES_128_GCM_IV12_TAG16_HKDF_SHA256_ECDSA_P256),
212        [0x03, 0x46] => Ok(&ESDK_ALG_AES_192_GCM_IV12_TAG16_HKDF_SHA384_ECDSA_P384),
213        [0x03, 0x78] => Ok(&ESDK_ALG_AES_256_GCM_IV12_TAG16_HKDF_SHA384_ECDSA_P384),
214        [0x04, 0x78] => Ok(&ESDK_ALG_AES_256_GCM_HKDF_SHA512_COMMIT_KEY),
215        [0x05, 0x78] => Ok(&ESDK_ALG_AES_256_GCM_HKDF_SHA512_COMMIT_KEY_ECDSA_P384),
216        [0x67, 0x00] => Ok(&DBE_ALG_AES_256_GCM_HKDF_SHA512_COMMIT_KEY_SYMSIG_HMAC_SHA384),
217        [0x67, 0x01] => {
218            Ok(&DBE_ALG_AES_256_GCM_HKDF_SHA512_COMMIT_KEY_ECDSA_P384_SYMSIG_HMAC_SHA384)
219        }
220        _ => Err(mpl_err("Invalid BinaryId")),
221    }

227const fn valid_suite(a: &AlgorithmSuite) -> bool {
228    true
229    /*
230    && KeyDerivationAlgorithm?(a.kdf)
231    && CommitmentDerivationAlgorithm?(a.commitment)
232    && EdkWrappingAlgorithm?(a.edkWrapping)
233
234    // If there is a KDF, the output length MUST match the encrypt length
235    && (a.kdf.HKDF? ==>
236          && a.kdf.HKDF.outputKeyLength == a.encrypt.AES_GCM.keyLength)
237       // If there is a signature, there MUST be a KDF
238    && (a.signature.ECDSA? ==> a.kdf.HKDF?)
239       // If there is commitment, the KDF MUST match
240    && (a.commitment.HKDF? ==>
241          && a.commitment.HKDF.saltLength == 32
242          && a.commitment == a.kdf)
243       // If there is a IntermediateKeyWrapping, the KDFs MUST match
244    && (a.edkWrapping.IntermediateKeyWrapping? ==>
245          && a.kdf.HKDF?
246          && a.edkWrapping.IntermediateKeyWrapping.keyEncryptionKeyKdf == a.kdf
247          && a.edkWrapping.IntermediateKeyWrapping.macKeyKdf == a.kdf)
248       // If there is a KDF and no commitment then salt MUST be 0
249    && (a.kdf.HKDF? && a.commitment.None? ==> a.kdf.HKDF.saltLength == 0)
250       //= aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-signature-settings
251       //= type=implication
252       //# An algorithm suite with a symmetric signature algorithm MUST use [intermediate key wrapping](#intermediate-key-wrapping).
253       //
254       // If the algorithm suite includes an symmetric signature algorithm:
255       //= aws-encryption-sdk-specification/framework/algorithm-suites.md#symmetric-signature-algorithm
256       //# - The algorithm suite MUST also use [Intermediate Key Wrapping](#intermediate-key-wrapping).
257    && (!a.symmetricSignature.None? ==>
258          && a.edkWrapping.IntermediateKeyWrapping?)
259          */
260}

453pub fn valid_algorithm_suite_info(suite: &AlgorithmSuite) -> Result<(), Error> {
454    let valid = match suite.id {
455        AlgorithmSuiteId::Esdk(a) => valid_esdk_suite(suite, a),
456        AlgorithmSuiteId::Dbe(a) => valid_dbe_suite(suite, a),
457    };
458    if valid {
459        Ok(())
460    } else {
461        Err(err(
462            ErrorKind::InvalidAlgorithmSuiteInfo,
463            "Invalid AlgorithmSuiteInfo",
464        ))
465    }
466}
