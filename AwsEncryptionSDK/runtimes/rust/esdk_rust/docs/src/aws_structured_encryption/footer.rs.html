20#[allow(unused)]
21pub(crate) fn foo(
22    enc: &aws_mpl_rs::EncryptionMaterials,
23    dec: &aws_mpl_rs::DecryptionMaterials,
24    data: &[CanonCryptoItem],
25    header: &[u8],
26    edks: &[aws_mpl_rs::EncryptedDataKey],
27) -> Result<(), Error> {
28    let footer = Footer::new(enc, data, header)?;
29    let footer_attribute = footer.make_terminal();
30    let footer2 = Footer::deserialize(&footer_attribute.value, true)?;
31    footer2.validate(dec, edks, data, header);
32    Ok(())
33}

35const RECIPIENT_TAG_SIZE: usize = 48;
36const SIGNATURE_SIZE: usize = 103;
37type RecipientTag = [u8; RECIPIENT_TAG_SIZE];
38type Signature = [u8; SIGNATURE_SIZE];
39const ENCRYPTED: &[u8] = b"ENCRYPTED";
40const PLAINTEXT: &[u8] = b"PLAINTEXT";

42//= specification/structured-encryption/footer.md#footer-format
43//= type=implication
44//# The [Terminal Value](./structures.md#terminal-value) of the footer MUST be
45// | Field | Length (bytes) | Interpreted as |
46// | ----- | -------------- | -------------- |
47// | [Recipient Tags](#recipient-tags) | Variable. 48 bytes per Encrypted Data Key in the header | Bytes |
48// | [Signature](#signature) | 0 or 103 | Signature, if signatures are enabled |
49pub(crate) struct Footer {
50    tags: Vec<RecipientTag>,
51    sig: Option<Signature>,
52}
53
54impl Footer {
55    fn serialize(&self, out: &mut Vec<u8>) {
56        serialize_tags(out, &self.tags);
57        serialize_sig(out, self.sig);
58    }

60    pub(crate) fn make_terminal(&self) -> StructuredDataTerminal {
61        //= specification/structured-encryption/encrypt-path-structure.md#footer-field
62        //= type=implication
63        //# The Footer Field TypeID MUST be 0xFFFF
64        //= specification/structured-encryption/encrypt-path-structure.md#footer-field
65        //= type=implication
66        //# The Footer Field Value MUST be the serialized [footer](footer.md).
67        let mut out: Vec<_> = Vec::new();
68        self.serialize(&mut out);
69        value_to_data(out, BINARY)
70    }

72    pub(crate) fn validate(
73        &self,
74        mat: &aws_mpl_rs::DecryptionMaterials,
75        edks: &[aws_mpl_rs::EncryptedDataKey],
76        data: &[CanonCryptoItem],
77        header: &[u8],
78    ) -> Result<(), Error>
79// requires Materials.DecryptionMaterialsWithPlaintextDataKey(mat)
80        // requires ValidSuite(mat.algorithmSuite)
81        // requires Header.ValidEncryptionContext(mat.encryptionContext)
82        // //= specification/structured-encryption/decrypt-path-structure.md#verify-signatures
83        // //= type=implication
84        // //# The number of [HMACs in the footer](./footer.md#hmacs)
85        // //# MUST be the number of [Encrypted Data Keys in the header](./header.md#encrypted-data-keys).
86        // ensures ret.Success? ==>
87        //           |edks| == |tags|
88    {
89        need(
90            edks.len() == self.tags.len(),
91            "There are a different number of recipient tags in the stored header than there are in the decryption materials.",
92        )?;
93        let canonical_hash = canon_hash(data, header, &mat.encryption_context);
94        let hash = aws_mpl_primitives::hmac(
95            get_hmac_alg(mat.algorithm_suite.symmetric_signature),
96            &mat.symmetric_signing_key.as_ref().unwrap().0,
97            &canonical_hash,
98        );
99        need(hash.len() == 48, "Bad hash length")?;
100
101        let mut found_tag = false;
102        for tag in &self.tags {
103            //= specification/structured-encryption/footer.md#recipient-tag-verification
104            //# Recipient Tag comparisons MUST be constant time operations.
105            if aws_mpl_primitives::constant_time_equal(&hash, tag) {
106                found_tag = true;
107                break;
108            }
109        }
110
111        //= specification/structured-encryption/footer.md#recipient-tag-verification
112        //# Verification MUST fail unless at least one of the [Recipient Tags](#recipient-tags)
113        //# matches a calculated recipient tag using the provided symmetricSigningKey.
114        need(
115            found_tag,
116            "Signature of record does not match the signature computed when the record was encrypted.",
117        )?;
118
119        // need(self.sig.is_some() == mat.algorithm_suite.signature.Ecdsa.is_some(), E("Internal error. Signature both does and does not exist."))?;
120        //= specification/structured-encryption/footer.md#signature-verification
121        //# If the footer contains a signature, this signature MUST be verified using the
122        //# [asymmetric signature algorithm](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-signature-settings)
123        //# indicated by the algorithm suite.
124        if let Some(sig) = self.sig.as_ref() {
125            aws_mpl_primitives::ecdsa_verify(
126                get_ecdsa_alg(mat.algorithm_suite.signature),
127                &mat.verification_key.as_ref().unwrap().0,
128                &canonical_hash,
129                sig,
130            )?;
131        }
132        Ok(())
133    }

135    pub(crate) fn deserialize(data: &[u8], has_sig: bool) -> Result<Self, Error> {
136        if has_sig {
137            need(
138                data.len() >= RECIPIENT_TAG_SIZE + SIGNATURE_SIZE,
139                "Footer too short.",
140            )?;
141            need(
142                (data.len() - SIGNATURE_SIZE).is_multiple_of(RECIPIENT_TAG_SIZE),
143                "Mangled signed footer has strange size",
144            )?;
145            Ok(Self {
146                tags: gather_tags(&data[..data.len() - SIGNATURE_SIZE]),
147                sig: Some(data[data.len() - SIGNATURE_SIZE..].try_into().unwrap()),
148            })
149        } else {
150            need(
151                data.len().is_multiple_of(RECIPIENT_TAG_SIZE),
152                "Mangled unsigned footer has strange size",
153            )?;
154            need(data.len() >= RECIPIENT_TAG_SIZE, "Footer too short.")?;
155            Ok(Self {
156                tags: gather_tags(data),
157                sig: None,
158            })
159        }
160    }

162    // return the footer value for the StructuredData
163    pub(crate) fn new(
164        mat: &aws_mpl_rs::EncryptionMaterials,
165        data: &[CanonCryptoItem],
166        header: &[u8],
167    ) -> Result<Self, Error>
168// requires ValidSuite(mat.algorithmSuite)
169        // requires Materials.EncryptionMaterialsHasPlaintextDataKey(mat)
170        // requires Header.ValidEncryptionContext(mat.encryptionContext)
171
172        // ensures (ret.Success? && mat.algorithmSuite.signature.ECDSA?) ==>
173        //           //= specification/structured-encryption/footer.md#signature
174        //           //= type=implication
175        //           //# The `signature`, if it exists, MUST be calculated using the
176        //           //# [asymmetric signature algorithm](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-signature-settings)
177        //           //# indicated by the algorithm suite.
178        //           && var history = client.History.ECDSASign;
179        //           && 0 < |history|
180        //           && var signInput = Seq.Last(history).input;
181        //           && signInput.signatureAlgorithm == mat.algorithmSuite.signature.ECDSA.curve
182    {
183        let canonical_hash = canon_hash(data, header, &mat.encryption_context);
184        let mut tags: Vec<RecipientTag> = Vec::new();
185        for key in &mat.symmetric_signing_keys {
186            //= specification/structured-encryption/footer.md#recipient-tags
187            //# the Recipient Tag MUST be MUST be calculated over the [Canonical Hash](#canonical-hash)
188            //# using the [symmetric signature algorithm](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-signature-settings)
189            //# indicated in the algorithm suite,
190            //# and the
191            //# [symmetric signing keys](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/structures.md#symmetric-signing-keys)
192            //# in the encryption materials.
193            let hmac = aws_mpl_primitives::hmac(
194                get_hmac_alg(mat.algorithm_suite.symmetric_signature),
195                &key.0,
196                &canonical_hash,
197            );
198            // :- Need(|hash| as u64 == 48, E("Bad hash length"));
199
200            //= specification/structured-encryption/footer.md#recipient-tags
201            //# the HMAC values MUST have the same order as the
202            //# [symmetric signing keys](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/structures.md#symmetric-signing-keys)
203            //# used to calculate them.
204            tags.push((&hmac[..]).try_into().unwrap());
205        }
206        //= specification/structured-encryption/footer.md#recipient-tags
207        //= type=implication
208        //# There MUST be one Recipient Tag for each Encrypted Data Key in the [header](./header.md#encrypted-data-keys)
209        debug_assert!(tags.len() == mat.encrypted_data_keys.len());
210
211        match mat.algorithm_suite.signature {
212            aws_mpl_rs::suites::SignatureAlgorithm::Ecdsa(x) => {
213                //= specification/structured-encryption/footer.md#signature
214                //# The `signature`, if it exists, MUST be calculated over the [Canonical Hash](#canonical-hash),
215                //# using the asymmetric signing key in the encryption materials.
216                let sig = aws_mpl_primitives::ecdsa_sign(
217                    x,
218                    &mat.signing_key.as_ref().unwrap().0,
219                    &canonical_hash,
220                )?;
221                need(
222                    sig.len() == SIGNATURE_SIZE,
223                    format!(
224                        "Signature is {} bytes, should have been {SIGNATURE_SIZE} bytes.",
225                        sig.len()
226                    ),
227                )?;
228                Ok(Self {
229                    tags,
230                    sig: Some(sig.try_into().unwrap()),
231                })
232            }
233            aws_mpl_rs::suites::SignatureAlgorithm::None => Ok(Self { tags, sig: None }),
234            _ => panic!(),
235        }
236    }
237}

239fn get_ecdsa_alg(
240    x: aws_mpl_rs::suites::SignatureAlgorithm,
241) -> aws_mpl_primitives::EcdsaSignatureAlgorithm {
242    match x {
243        aws_mpl_rs::suites::SignatureAlgorithm::Ecdsa(x) => x,
244        _ => panic!(),
245    }
246}
247fn get_hmac_alg(
248    x: aws_mpl_rs::suites::SymmetricSignatureAlgorithm,
249) -> aws_mpl_primitives::DigestAlg {
250    match x {
251        aws_mpl_rs::suites::SymmetricSignatureAlgorithm::Hmac(x) => x,
252        _ => panic!(),
253    }
254}

256fn get_canonical_encrypted_field(
257    data: &mut Vec<u8>,
258    canonical_field_name: &[u8],
259    value: &StructuredDataTerminal,
260) {
261    //= specification/structured-encryption/footer.md#canonical-encrypted-field
262    //= type=implication
263    //# The canonical form of an encrypted field MUST be
264    //# | Field | Length (bytes) | Interpreted as |
265    //# | ----- | -------------- | -------------- |
266    //# | The [canonical path](./header.md#canonical-path) of the field name | Variable | Bytes |
267    //# | encrypted data length - 2 | 8 | 64-bit integer |
268    //# | "ENCRYPTED" | 9 | Literal Ascii text |
269    //# | TypeID | 2 | the type ID of the unencrypted Terminal |
270    //# | value | Variable | the encrypted Terminal value |
271
272    data.extend_from_slice(canonical_field_name);
273    write_u64(data, value.value.len() as u64 - 2);
274    data.extend_from_slice(ENCRYPTED);
275    data.extend_from_slice(&value.value);
276}
277
278fn get_canonical_plaintext_field(
279    data: &mut Vec<u8>,
280    canonical_field_name: &[u8],
281    value: &StructuredDataTerminal,
282)
283//= specification/structured-encryption/footer.md#canonical-plaintext-field
284//= type=implication
285//# The canonical form of a plaintext field MUST be
286//# | Field | Length (bytes) | Interpreted as |
287//# | ----- | -------------- | -------------- |
288//# | The [canonical path](./header.md#canonical-path) of the field name | Variable | Bytes |
289//# | data length | 8 | 64-bit integer |
290//# | "PLAINTEXT" | 9 | Literal Ascii text |
291//# | TypeID | 2 | the type ID of the Terminal |
292//# | value | Variable | the Terminal value |
293{
294    data.extend_from_slice(canonical_field_name);
295    write_u64(data, value.value.len() as u64);
296    data.extend_from_slice(PLAINTEXT);
297    data.extend_from_slice(&value.type_id);
298    data.extend_from_slice(&value.value);
299}

301// Given a key value pair, return the canonical value for use in the footer checksum calculations
302fn get_canonical_item(data: &mut Vec<u8>, item: &CanonCryptoItem) {
303    if item.action == CryptoAction::EncryptAndSign {
304        get_canonical_encrypted_field(data, &item.key, &item.data);
305    } else {
306        get_canonical_plaintext_field(data, &item.key, &item.data);
307    }
308}
309
310fn canon_content(data: &mut Vec<u8>, items: &[CanonCryptoItem]) {
311    for item in items {
312        if item.action != CryptoAction::DoNothing {
313            get_canonical_item(data, item);
314        }
315    }
316}

318fn canon_record(
319    data: &mut Vec<u8>,
320    items: &[CanonCryptoItem],
321    header: &[u8],
322    enc: &EncryptionContext,
323)
324//= specification/structured-encryption/footer.md#canonical-record
325//= type=implication
326//# The canonical form of a record MUST be
327//# | Field | Length (bytes) | Interpreted as |
328//# | ----- | -------------- | -------------- |
329//# | header | Variable | The full serialized header with commitment |
330//# | AAD Length | 8 | 64-bit integer, the length of the following AAD data |
331//# | AAD | Variable | The serialization of the Encryption Context from the Encryption Materials |
332//# | Field Data | Variable | For each [signed field](#signed-fields), ordered lexicographically by [canonical path](./header.md#canonical-path), the [canonical field](#canonical-field).
333{
334    data.extend_from_slice(header);
335    // FIXME -- we can calculate the length without having to serialize the context
336    let mut aad = Vec::new();
337    header::serialize_context(&mut aad, enc);
338    write_u64(data, aad.len() as u64);
339    data.append(&mut aad);
340    canon_content(data, items);
341}

343fn canon_hash(items: &[CanonCryptoItem], header: &[u8], enc: &EncryptionContext) -> Vec<u8>
344//= specification/structured-encryption/footer.md#hash-calculation
345        //= type=implication
346        //# The canonical hash of a record MUST be the SHA384 of the canonical form of the record.
347{
348    let mut data = Vec::new();
349    canon_record(&mut data, items, header, enc);
350    aws_mpl_primitives::digest(aws_mpl_primitives::DigestAlg::Sha384, &data)
351}
352
353fn serialize_tags(data: &mut Vec<u8>, tags: &[RecipientTag]) {
354    for tag in tags {
355        data.extend_from_slice(tag);
356    }
357}
358
359fn serialize_sig(data: &mut Vec<u8>, sig: Option<Signature>) {
360    if let Some(s) = sig {
361        data.extend_from_slice(&s);
362    }
363}
364
365fn gather_tags(data: &[u8]) -> Vec<RecipientTag> {
366    data.chunks_exact(RECIPIENT_TAG_SIZE)
367        .map(|chunk| chunk.try_into().unwrap())
368        .collect()
