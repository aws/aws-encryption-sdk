20// need to return an array, not a Vec
21fn field_key(hkdf_output: &[u8], offset: u32) -> Result<Vec<u8>, Error>
22//= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
23//= type=implication
24//# The `FieldKey` for a given key and offset MUST be the first 44 bytes
25//# of the aes256ctr_stream
26//# of the `FieldRootKey` and the `FieldKeyNonce` of three times the given offset.
27{
28    let key = aws_mpl_primitives::ctr_stream(
29        &field_key_nonce(offset * 3),
30        hkdf_output,
31        KEY_SIZE + NONCE_SIZE,
32    )?;
33    Ok(key)
34}
35
36const AWS_DBE_FIELD: &[u8] = b"AwsDbeField";
37const LABEL_COMMITMENT_KEY: &[u8] = b"AWS_DBE_COMMIT_KEY";
38const LABEL_ENCRYPTION_KEY: &[u8] = b"AWS_DBE_DERIVE_KEY";
39
40fn field_key_nonce(offset: u32) -> Vec<u8>
41//= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
42    //= type=implication
43    //# The `FieldKeyNonce` for a given offset MUST be 16 bytes comprised of
44    //# | Field         | Length   | Interpretation |
45    //# | ------------- | -------- | -------------- |
46    //# | "AwsDbeField" | 11       | Literal Ascii String |
47    //# | 0x2c          | 1        | 44, the length of the eventual FieldKey |
48    //# | offset        | 4        | 32 bit integer representation of offset |
49{
50    let mut ret = Vec::new();
51    ret.extend_from_slice(AWS_DBE_FIELD);
52    ret.push((KEY_SIZE + NONCE_SIZE) as u8);
53    write_u32(&mut ret, offset);
54    ret
55}

57fn get_aes_alg(x: Encrypt) -> Result<aws_mpl_primitives::AesGcm, Error> {
58    match x {
59        Encrypt::AesGcm(e) => Ok(e),
60        _ => Err("Must be AesGcm".into()),
61    }
62}
63
64fn get_hkdf_alg(x: &DerivationAlgorithm) -> Result<aws_mpl_primitives::DigestAlg, Error> {
65    match x {
66        DerivationAlgorithm::Hkdf(h) => Ok(h.hmac),
67        _ => Err("Must be Hkdf".into()),
68    }
69}
70fn get_hkdf_outlen(x: &DerivationAlgorithm) -> Result<usize, Error> {
71    match x {
72        DerivationAlgorithm::Hkdf(h) => Ok(h.output_key_length as usize),
73        _ => Err("Must be Hkdf".into()),
74    }
75}
76// suitable for header field
77fn get_commit_key(alg: &AlgorithmSuite, key: &Key, msg_id: &MessageID) -> Result<Vec<u8>, Error>
78//= specification/structured-encryption/header.md#commit-key
79              //= type=implication
80              //# The calculated Commitment Key MUST have length equal to the
81              //# [algorithm suite's encryption key length](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-encryption-settings).
82
83              //= specification/structured-encryption/header.md#commit-key
84              //= type=implication
85              //# The HKDF used to calculate the Commitment Key MUST be the
86              //# [Commit Key KDF](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-commit-key-derivation-settings)
87              //# indicated by the algorithm suite.
88
89              //= specification/structured-encryption/header.md#commit-key
90              //= type=implication
91              //# The `info` used for the HKDF function MUST be
92              //# | Field                | Length   |
93              //# | -------------------- | -------- |
94              //# | "AWS_DBE_COMMIT_KEY" | 18       |
95              //# | Message ID           | 32       |
96
97              //= specification/structured-encryption/header.md#commit-key
98              //= type=implication
99              //# The HKDF calculation MUST use a supplied key, no salt, and an `info` as described above.
100{
101    let mut commit_key = vec![0; get_hkdf_outlen(&alg.commitment)?];
102    aws_mpl_primitives::hkdf_no_salt(
103        get_hkdf_alg(&alg.commitment)?,
104        key,
105        &[LABEL_COMMITMENT_KEY, msg_id],
106        &mut commit_key,
107    )?;
108    Ok(commit_key)
109}

131// Encrypt or Decrypt a StructuredDataMap
132fn crypt(
133    mode: EncryptionSelector,
134    alg: &AlgorithmSuite,
135    key: &Key,
136    head: &header::PartialHeader,
137    data: &[CanonCryptoItem],
138) -> Result<Vec<CanonCryptoItem>, Error>
139//= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
140            //= type=implication
141            //# The HKDF algorithm used to calculate the Field Root Key MUST be the
142            //# [Encryption Key KDF](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-encryption-key-derivation-settings)
143            //# indicated by the algorithm suite, using a provided plaintext data key, no salt,
144            //# and an info as calculated [above](#calculate-info)
145
146            //= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
147            //= type=implication
148            //# The `FieldRootKey` MUST be generated with the plaintext data key in the encryption materials
149            //# and the Message ID generated for this Encrypted Structured Data.
150
151            //= specification/structured-encryption/encrypt-path-structure.md#calculate-info
152            //= type=implication
153            //# The `info` used for the HKDF function MUST be
154            //# | Field                | Length   |
155            //# | -------------------- | -------- |
156            //# | "AWS_DBE_DERIVE_KEY" | 18       |
157            //# | Message ID           | 32       |
158{
159    //= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
160    //# The `FieldRootKey` MUST be generated with the plaintext data key in the encryption materials
161    //# and the Message ID generated for this Encrypted Structured Data.
162    let mut field_root_key = vec![0; get_hkdf_outlen(&alg.kdf)?];
163    aws_mpl_primitives::hkdf_no_salt(
164        get_hkdf_alg(&alg.kdf)?,
165        key,
166        &[LABEL_ENCRYPTION_KEY, &head.msg_id],
167        &mut field_root_key,
168    )?;
169
170    //= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
171    //= type=implication
172    //# The calculated Field Root MUST have length equal to the
173    //# [algorithm suite's encryption key length](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-encryption-settings).
174    crypt_list(mode, alg, &field_root_key.try_into().unwrap(), data)
175}

177// Encrypt or Decrypt each entry in keys, putting results in output
178
179fn crypt_list(
180    mode: EncryptionSelector,
181    alg: &AlgorithmSuite,
182    field_root_key: &Key,
183    data: &[CanonCryptoItem],
184) -> Result<Vec<CanonCryptoItem>, Error> {
185    let mut result: Vec<CanonCryptoItem> = Vec::with_capacity(data.len());
186    let mut pos: u32 = 0;
187    for item in data {
188        if item.action == CryptoAction::EncryptAndSign {
189            let new_terminal = if mode == EncryptionSelector::DoEncrypt {
190                encrypt_terminal(alg, field_root_key, pos, &item.key, &item.data)?
191            } else {
192                decrypt_terminal(alg, field_root_key, pos, &item.key, &item.data)?
193            };
194            pos += 1;
195            let new_item = CanonCryptoItem {
196                data: new_terminal,
197                ..item.clone()
198            };
199            result.push(new_item);
200        } else {
201            result.push(item.clone());
202        }
203    }
204    Ok(result)
205}

207// Encrypt a single Terminal
208fn encrypt_terminal(
209    alg: &AlgorithmSuite,
210    field_root_key: &Key,
211    offset: u32,
212    path: &[u8],
213    data: &StructuredDataTerminal,
214) -> Result<StructuredDataTerminal, Error>
215//= specification/structured-encryption/encrypt-path-structure.md#terminal-data-encryption
216              //= type=implication
217              //# The output encrypted Terminal Data MUST have a [Terminal Type Id](./structures.md#terminal-type-id)
218              //# equal `0xFFFF`.
219
220              //= specification/structured-encryption/encrypt-path-structure.md#terminal-data-encryption
221              //= type=implication
222              //# The output encrypted Terminal Data MUST have a [Terminal Value](./structures.md#terminal-value)
223              //# with the following serialization:
224                 // | Field                      | Length   |
225                 // | -------------------------- | -------- |
226                 // | Terminal Type Id           | 2        |
227                 // | Encrypted Terminal Value   | Variable |
228
229                 //= specification/structured-encryption/encrypt-path-structure.md#terminal-type-id
230                 //= type=implication
231                 //# Terminal Type Id MUST equal the input Terminal Data's Terminal Type Id.
232
233                 //= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
234              //= type=implication
235              //# The `Cipherkey` MUST be the first 32 bytes of the `FieldKey`
236
237              //= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
238                 //= type=implication
239                 //# The `Nonce` MUST be the remaining 12 bytes of the `FieldKey`
240{
241    let data_key: Vec<u8> = field_key(field_root_key, offset)?;
242    //= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
243    //# The `Cipherkey` MUST be the first 32 bytes of the `FieldKey`
244    let encryption_key: Key = data_key[..KEY_SIZE].try_into().unwrap();
245    //= specification/structured-encryption/encrypt-path-structure.md#calculate-cipherkey-and-nonce
246    //# The `Nonce` MUST be the remaining 12 bytes of the `FieldKey`
247    let nonce: Nonce = data_key[KEY_SIZE..].try_into().unwrap();
248    let value = &data.value;
249
250    //= specification/structured-encryption/encrypt-path-structure.md#encrypted-terminal-value
251    //# The Encrypted Terminal Value MUST be derived according to the following encryption:
252    // - The encryption algorithm used is the
253    //   [encryption algorithm](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-encryption-settings)
254    //   indicated in the algorithm suite.
255    // - The AAD is the [canonical path](./header.md#canonical-path) for this Terminal Data
256    // - The [Cipherkey and Nonce](#calculate-cipherkey-and-nonce) are as calculated above.
257    // - The plaintext is the [Terminal Value](./structures.md#terminal-value) for this Terminal Data.
258
259    // TODO - make new aes_encrypt that takes &mut[u8]
260    // let mut cipher_text = vec![0; value.len() + TYPE_ID_LEN + AUTH_TAG_SIZE];
261    let mut cipher_text = Vec::with_capacity(value.len() + AUTH_TAG_SIZE);
262    aws_mpl_primitives::aes_encrypt(
263        get_aes_alg(alg.encrypt)?,
264        &nonce,
265        &encryption_key,
266        value,
267        path,
268        &mut cipher_text,
269    )?;
270    let mut result = data.type_id.to_vec();
271    result.extend_from_slice(&cipher_text);
272    Ok(value_to_data(result, BINARY))
273}

275// Can we take data by value, and return it?
276// Decrypt a single Terminal
277fn decrypt_terminal(
278    alg: &AlgorithmSuite,
279    field_root_key: &Key,
280    offset: u32,
281    path: &[u8],
282    data: &StructuredDataTerminal,
283) -> Result<StructuredDataTerminal, Error> {
284    let data_key: Vec<u8> = field_key(field_root_key, offset)?;
285    let encryption_key: Key = data_key[..KEY_SIZE].try_into().unwrap();
286    let nonce: Nonce = data_key[KEY_SIZE..].try_into().unwrap();
287    let value = &data.value;
288    need((AUTH_TAG_SIZE + 2) <= value.len(), "cipherTxt too short.")?;
289
290    //= specification/structured-encryption/decrypt-path-structure.md#terminal-data-decryption
291    //# The input [Terminal Value](./structures.md#terminal-value) MUST be deserialized as follows:
292    // | Field                      | Length   |
293    // | -------------------------- | -------- |
294    // | Terminal Type Id           | 2        |
295    // | Encrypted Terminal Value   | Variable |
296
297    //= specification/structured-encryption/decrypt-path-structure.md#terminal-data-decryption
298    //# The output Terminal Data MUST have a [Terminal Value](./structures.md#terminal-type-id)
299    //# equal to the following decryption:
300    // - The decryption algorithm used is the
301    //   [encryption algorithm](../../submodules/MaterialProviders/aws-encryption-sdk-specification/framework/algorithm-suites.md#algorithm-suites-encryption-settings)
302    //   indicated in the algorithm suite.
303    // - The AAD is the [canonical path](./header.md#canonical-path) for this Terminal Data
304    // - The Cipherkey and Nonce are as calculate [above](#calculate-cipherkey-and-nonce).
305    // - The ciphertext is the deserialized Encrypted Terminal Value.
306    let mut plain_text = vec![0; value.len() - TYPE_ID_LEN - AUTH_TAG_SIZE];
307    aws_mpl_primitives::aes_decrypt(
308        get_aes_alg(alg.encrypt)?,
309        &encryption_key,
310        &value[TYPE_ID_LEN..value.len() - AUTH_TAG_SIZE],
311        &value[value.len() - AUTH_TAG_SIZE..],
312        &nonce,
313        path,
314        &mut plain_text,
315    )?;
316
317    Ok(value_to_data(
318        plain_text,
319        value[..TYPE_ID_LEN].try_into().unwrap(),
320    ))
