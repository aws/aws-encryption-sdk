33pub(crate) fn write_header_body(w: &mut dyn SafeWrite, body: &HeaderBody) -> Result<(), Error> {
34    match body {
35        HeaderBody::V1Body(x) => write_v1_header_body(w, x),
36        HeaderBody::V2Body(x) => write_v2_header_body(w, x),
37    }
38}

40pub(crate) fn read_header_body(
41    r: &mut dyn SafeRead,
42    max_edks: Option<std::num::NonZeroUsize>,
43    raw: &mut dyn SafeWrite,
44) -> Result<HeaderBody, Error> {
45    let version = read_msg_format_version(r, raw)?;
46
47    let result = match version {
48        MessageFormatVersion::V1 => {
49            let body = read_v1_header_body(r, max_edks, raw)?;
50            HeaderBody::V1Body(body)
51        }
52        MessageFormatVersion::V2 => {
53            let body = read_v2_header_body(r, max_edks, raw)?;
54            HeaderBody::V2Body(body)
55        }
56    };
57
58    match result.content_type() {
59        ContentType::Framed => {
60            if result.frame_length() == 0 {
61                return ser_err("Frame length must be positive if content is framed");
62            }
63        }
64        ContentType::NonFramed => {
65            if result.frame_length() != 0 {
66                return ser_err("Frame length must be zero if content is non-framed");
67            }
68        }
69    }
70    Ok(result)
71}

73pub(crate) const fn header_version_supports_commitment(
74    suite: &AlgorithmSuite,
75    body: &HeaderBody,
76) -> bool {
77    match (suite.commitment, body) {
78        (DerivationAlgorithm::Hkdf(header), HeaderBody::V2Body(body)) => {
79            body.suite_data.len() == header.output_key_length as usize
80        }
81        (_, HeaderBody::V1Body(_)) => true,
82        _ => false,
83    }
